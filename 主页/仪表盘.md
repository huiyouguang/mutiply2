```dataviewjs
    // 标签可点击：搜索该标签

                style="display:inline-block; background:${theme.tagBg}; color:${theme.text}; 
                       padding:6px 12px; border-radius:20px; text-decoration:none; font-size:0.9em;">
                #${tag} <span style="color:${theme.accent};">(${count})</span>
             </a>`;
}

html += `</div></div>`;
dv.paragraph(html);
```
